create database fastBidingdb;


create table proposalAdmin
(
pid int primary key auto_increment,
client varchar(20),
email      varchar(20),
location   varchar(20),
password   varchar(20),
);
alter table proposalAdmin add column country varchar(20), state varchar(20),
 companyRegistrationNo varchar(20); 


create table vendors 
(
vid int primary key,
vname  varchar(20),
email  varchar(20),
country varchar(20),
state varchar(20),
address varchar(20),
password varchar(20)
);

alter table vendor add column companyRegistrationNum varchar(20), gstNum varchar(20);


create table siteAdmin
(
sid int primary key auto_increment,
sname varchar(20),
email varchar(20),
password varchar(20)
);

create table projects
(
pjId int primary key auto_increment,
pTitle varchar(20),
pCost varchar(20),
pSummary varchar(200),
pDetailsDocLink varchar(20),
type varchar(20) 
pAdmin int,
foreign key (pAdmin) references proposalAdmin(pid) 
);


create table proposals 
(
preId int primary key auto_increment,
pjId int unique,
prerequisite varchar(100),
startDate date,
endDate date,
);
